Test if a point is enclosed:

encloses :: BspTree -> Vector -> Bool
encloses (BspTree [] _ _) _ = False
encloses tree vertex = (BspTree (t:_) front back) =
    case vectorSideOfPlane (triToPlane t) v of
        FRONT -> case front of
                      (BspTree [] _ _) -> False
                      a -> encloses front vertex
        BACK -> case back of
                      (BspTree [] _ _) -> True
                      a -> encloses back vertex
        COPLANAR -> case back of
                      (BspTree [] _ _) -> True
                      a -> encloses back vertex
